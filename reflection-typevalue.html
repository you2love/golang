← 返回 reflection

reflect.Type 和 reflect.Value

package main

import (
    "fmt"
    "reflect"
)

func main() {
    var x int = 42

    // 获取类型和值
    t := reflect.TypeOf(x)
    v := reflect.ValueOf(x)

    fmt.Println("类型:", t)
    fmt.Println("值:", v)
    fmt.Println("类型种类:", t.Kind())
    fmt.Println("值的种类:", v.Kind())
}
