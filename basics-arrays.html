数组

package main

import "fmt"

func main() {
    // 声明数组
    var arr1 [5]int
    arr1[0] = 1
    arr1[1] = 2

    // 数组初始化
    arr2 := [5]int{1, 2, 3, 4, 5

    // 让编译器计算长度
    arr3 := [...]int{1, 2, 3}

    fmt.Println("数组1:", arr1)
    fmt.Println("数组2:", arr2)
    fmt.Println("数组3:", arr3)
    fmt.Println("数组2长度:", len(arr2))
}
