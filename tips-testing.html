测试技巧

1. 表驱动测试

package main

import (
    "testing"
)

type testCase struct {
    name     string
    input    int
    expected int
}

var testCases = []testCase{
    {"positive", 1, 1},
    {"zero", 0, 0},
    {"negative", -1, -1},
}

func TestAbs(t *testing.T) {
    for _, tc := range testCases {
        t.Run(tc.name, func(t *testing.T) {
            result := Abs(tc.input)
            if result != tc.expected {
                t.Errorf("expected %d, got %d", tc.expected, result)
            }
        })
    }
}
