性能优化

1. 字符串处理优化

package main

import (
    "fmt"
    "strings"
)

// ❌ 坏的做法：使用 + 拼接字符串
func badStringConcat(parts []string) string {
    result := ""
    for _, part := range parts {
        result += part  // 每次都创建新字符串
    }
    return result
}

// ✅ 好的做法：使用 strings.Builder
func goodStringConcat(parts []string) string {
    var builder strings.Builder
    builder.Grow(1024)  // 预分配空间
    for _, part := range parts {
        builder.WriteString(part)
    }
    return builder.String()
}

// ✅ 使用 fmt.Sprintf 格式化
func formatString(name string, age int) string {
    return fmt.Sprintf("Name: %s, Age: %d", name, age)
}

2. 切片和映射优化

package main

import "fmt"

// ✅ 预分配切片容量
func preallocateSlice(size int) []int {
    slice := make([]int, 0, size)
    for i := 0; i < size; i++ {
        slice = append(slice, i)
    }
    return slice
}

// ✅ 预分配映射容量
func preallocateMap(size int) map[string]int {
    m := make(map[string]int, size)
    for i := 0; i < size; i++ {
        m[fmt.Sprintf("key%d", i)] = i
    }
    return m
}

// ✅ 重用切片
func reuseSlice() {
    buf := make([]byte, 0, 1024)
    
    // 使用 buf
    data := buf[:100]
    
    // 重用
    buf = buf[:0]
    data = buf[:200}
}

3. 内存优化

package main

import (
    "fmt"
    "sync"
)

// ✅ 使用 sync.Pool 重用对象
var bufferPool = sync.Pool{
    New: func() interface{} {
        return make([]byte, 0, 1024)
    },
}

func useBufferFromPool() {
    buf := bufferPool.Get().([]byte)
    defer func() {
        buf = buf[:0]
        bufferPool.Put(buf)
    }()
    
    // 使用 buf
}
