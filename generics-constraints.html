类型约束

package main

import "fmt"

// 自定义类型约束
type Number interface {
    int | int64 | float64
}

func Sum[T Number](numbers []T) T {
    var total T
    for _, num := range numbers {
        total += num
    }
    return total
}

func main() {
    ints := []int{1, 2, 3, 4, 5}
    floats := []float64{1.1, 2.2, 3.3}

    fmt.Println("整数和:", Sum(ints))
    fmt.Println("浮点数和:", Sum(floats))
}
