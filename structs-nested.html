结构体嵌套

package main

import "fmt"

// 地址结构体
type Address struct {
    City    string
    Street  string
    ZipCode string
}

// 用户结构体
type User struct {
    Name    string
    Age     int
    Address // 嵌套结构体
}

func main() {
    user := User{
        Name: "张三",
        Age:  25,
        Address: Address{
            City:    "北京",
            Street:  "长安街",
            ZipCode: "100000",
        },
    }

    fmt.Println("用户:", user.Name)
    fmt.Println("城市:", user.Address.City)
    fmt.Println("街道:", user.Address.Street)
}
