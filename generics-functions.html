泛型函数

package main

import "fmt"

// 泛型函数：使用类型参数
func Min[T cmp.Ordered](x, y T) T {
    if x < y {
        return x
    }
    return y
}

func main() {
    // 使用整数
    fmt.Println(Min(5, 3))  // 3

    // 使用浮点数
    fmt.Println(Min(3.14, 2.71))  // 2.71

    // 使用字符串
    fmt.Println(Min("apple", "banana"))  // apple
}
